使用 mergeMap() 處理 Dependent API

2020-05-05

Promise 的 then() 具有 chain() 特性，可攤平 Nested Promise；若要攤平 Nested Observable 則要使用 mergeMap()。

Version

macOS Catalina 10.15.4
WebStorm 2020.1
Vue 2.6.11
Vue-rx 6.2
RxJS 6.5.5

Browser

顯示 category。

Data

{
  "title": "FP in JavaScript",
  "price": 100,
  "categoryId": 1,
}

http://localhost:3000/books/1 回傳為以上 object。

{
  "id": 1,
  "value": "FP"
}

http://localhost:3000/categories/1 回傳以上 object。

Promise Chain

<template>
  <div>
    {{ category }}
  </div>
</template>

<script>
import axios from 'axios'

let fetchBook = id => axios.get(`http://localhost:3000/books/${id}`)

let fetchCategory = id => axios.get(`http://localhost:3000/categories/${id}`)

let mounted = function() {
  fetchBook(1)
    .then(x => x.data.categoryId)
    .then(x => fetchCategory(x))
    .then(x => x.data.value)
    .then(x => this.category = x)
}

export default {
  name:'app',
  data:() => ({
    category:''
  }),
  mounted
}
</script>

10 行

let fetchBook = id => axios.get(`http://localhost:3000/books/${id}`)

由 fetchBook() 回傳指定 id 的 object，為 Promise 包含 categoryId property，將呼叫其他 API 取得 category。

12 行

let fetchCategory = id => axios.get(`http://localhost:3000/categories/${id}`)

由 fetchCategory$() 回傳指定 id 的 object，亦為 Promise 包含 value property。

15 行

fetchBook(1)
  .then(x => x.data.categoryId)
  .then(x => fetchCategory(x))
  .then(x => x.data.value)
  .then(x => this.category = x)

由於 Promise 的 then() 具有 chain() 特性，可將 Nested Promise 攤平，因此不會出現 Nested Callback。

Ramda

<template>
  <div>
    {{ category }}
  </div>
</template>

<script>
import axios from 'axios'
import { pipe, andThen, path } from 'ramda'

let fetchBook = id => axios.get(`http://localhost:3000/books/${id}`)

let fetchCategory = id => axios.get(`http://localhost:3000/categories/${id}`)

let mounted = function() {
  pipe(
    fetchBook,
    andThen(path(['data', 'categoryId'])),
    andThen(fetchCategory),
    andThen(path(['data', 'value'])),
    andThen(x => this.category = x)
  )(1)
}

export default {
  name:'app',
  data:() => ({
    category:''
  }),
  mounted
}
</script>

16 行

pipe(
  fetchBook,
  andThen(path(['data', 'categoryId'])),
  andThen(fetchCategory),
  andThen(path(['data', 'value'])),
  andThen(x => this.category = x)
)(1)

Ramda 的 pipe() 與 andThen() 基本上還是 Promise，只不過提供了 function 版本適合 function pipeline 與 point-free。

Nested Observable

<template>
  <div>
    {{ category }}
  </div>
</template>

<script>
import axios from 'axios'
import { from } from 'rxjs'
import { pluck } from 'rxjs/operators'

let fetchBook$ = id => from(axios.get(`http://localhost:3000/books/${id}`))

let fetchCategory$ = id => from(axios.get(`http://localhost:3000/categories/${id}`))

let mounted = function() {
  fetchBook$(1).pipe(
    pluck('data', 'categoryId'),
  ).subscribe(x => {
    fetchCategory$(x).pipe(
      pluck('data', 'value')
    ).subscribe(x => this.category = x)
  })
}

export default {
  name:'app',
  data:() => ({
    category:''
  }),
  mounted
}
</script>

12 行

let fetchBook$ = id => from(axios.get(`http://localhost:3000/books/${id}`))

let fetchCategory$ = id => from(axios.get(`http://localhost:3000/categories/${id}`))

API function 街回傳 Observable。

17 行

fetchBook$(1).pipe(
  pluck('data', 'categoryId'),
).subscribe(x => {
  fetchCategory$(x).pipe(
    pluck('data', 'value')
  ).subscribe(x => this.category = x)
})

由於 fetchCategory$() 依賴 fetchBook$() 先取得 categoryId，直覺寫法是在 fetchBook$() 的 subscribe() 內去呼叫 fetchCategory$()，最後在內層 subscribe() 寫入 category side effect。

<template>
  <div>
    {{ category$ }}
  </div>
</template>

<script>
import axios from 'axios'
import { from } from 'rxjs'
import { pluck, mergeMap } from 'rxjs/operators'

let fetchBook$ = id => from(axios.get(`http://localhost:3000/books/${id}`))

let fetchCategory$ = id => from(axios.get(`http://localhost:3000/categories/${id}`))

let category$ = fetchBook$(1).pipe(
  pluck('data', 'categoryId'),
  mergeMap(x => fetchCategory$(x)),
  pluck('data', 'value')
)

export default {
  name:'app',
  subscriptions:() => ({
    category$
  }),
}
</script>

16 行

let category$ = fetchBook$(1).pipe(
  pluck('data', 'categoryId'),
  mergeMap(x => fetchCategory$(x)),
  pluck('data', 'value')
)

在 subscribe() 內呼叫 API 產生新 Observable，也就是所謂 Nested Observable 時，適合重構成 mergeMap()，也稱為 flatMap()，可將 Nested Observable 攤平。

mergeMap()
將兩層 Observable 攤平成一層 Observable

Conclusion

  • Dependent API 在實務上常常出現，可使用 mergeMap() 攤平處理
