如何顯示 Nested Promise ?

2020-05-24

若 API 遵循 RESTful 精神設計，常會第一個 API 僅回傳 id，若要取得其值就要呼叫第二個 API，也就是所謂 N + 1 Query，這種實務上常見設計在 Vue 該如何顯示呢 ? 本文分別以 Aysnc Await、Promise Chain、Ramda 與 RxJS 四種寫法示範。

Version

macOS Catalina 10.15.4
WebStorm 2020.1
Vue 2.6.11
Ramda 0.27.0
RxJS 6.5.5

Browser

FP、FRP 與 JS 並不是從單一 API 取得，而是呼叫另外一個 API。

Double Side Effect

Async Await

<template>
  <div>
    <ul v-for="(x, i) in books" :key="i">
      <li>{{ x.title }} / {{ x.price }} / {{ categories[i] }}</li>
    </ul>
  </div>
</template>

<script>
let bookCategories = [
  { id: 1, category: 'FP' },
  { id: 2, category: 'FRP' },
  { id: 3, category: 'JS' }
]

let fetchCategory = async id =>
  bookCategories.find(x => x.id === id).category

let fetchBooks = async () => [
  { title: 'FP in JavaScript', price: 100, categoryId: 1 },
  { title: 'RxJS in Action', price: 200, categoryId: 2 },
  { title: 'Speaking JavaScript', price: 300, categoryId: 3 }
]

let mounted = async function() {
  this.books = await fetchBooks()

  for (let x of this.books)
    this.categories.push(await fetchCategory(x.categoryId))
}

export default {
  name:'App',
  data: () => ({
    books: [],
    categories: []
  }),
  mounted,
}
</script>

  { title: 'Speaking JavaScript', price: 300, categoryId: 3 }
]

模擬 Axios 從 API 回傳 Promise，故為 async function，由於遵守 RESTFul 設計，因此僅回傳 catetoryId，若要取得 category 必須再呼叫另外一個 API。

16 行

模擬另外一個 API 由 categoryId 取得 category。

26 行

this.books = await fetchBooks()

for (let x of this.books)
  this.categories.push(await fetchCategory(x.categoryId))

使用 books 與 categories 兩份 data，可直接對 data 做 for of loop 與 push。

第 3 行

<ul v-for="(x, i) in books" :key="i">
  <li>{{ x.title }} / {{ x.price }} / {{ categories[i] }}</li>
</ul>

目前有 books 與 categories 兩份 data，但 v-for 只能同時處理一份，故 categories 必須靠 index 存取。

這種處理方式的特點是每個 API 在 Vue 都有各自 data，v-for 只使用其中一份，另外一份由 index 存取，優點是 API 容易處理，缺點 HTML template 必須技巧性處理兩份 data

Promise Chain

<template>
  <div>
    <ul v-for="(x, i) in books" :key="i">
      <li>{{ x.title }} / {{ x.price }} / {{ categories[i] }}</li>
    </ul>
  </div>
</template>

<script>
let bookCategories = [
  { id: 1, category: 'FP' },
  { id: 2, category: 'FRP' },
  { id: 3, category: 'JS' }
]

let fetchCategory = async id =>
  bookCategories.find(x => x.id === id).category

let fetchBooks = async () => [
  { title: 'FP in JavaScript', price: 100, categoryId: 1 },
  { title: 'RxJS in Action', price: 200, categoryId: 2 },
  { title: 'Speaking JavaScript', price: 300, categoryId: 3 }
]

let mounted = function() {
  fetchBooks()
    .then(x => this.books = x)
    .then(x => Promise.all(x.map(y => fetchCategory(y.categoryId))))
    .then(x => this.categories = x)
}

export default {
  name:'App',
  data: () => ({
    books: [],
    categories: []
  }),
  mounted,
}
</script>

26 行

fetchBooks()
  .then(x => this.books = x)

因為是 double side effect，所以 fetchBooks() 得到資料後先寫入 books data，避免 Promise Chain 不方便傳遞資料的尷尬，順便將 books data 回傳給下一個 then() 供第二個 API 使用。

28 行

.then(x => Promise.all(x.map(y => fetchCategory(y.categoryId))))
.then(x => this.categories = x)

x.map(y => fetchCategory(y.categoryId)) 回傳為 Promise Array，故必須使用 Promise.all() 取得 Array 內部值，但會重新包成新 Promise 回傳。

Ramda

<template>
  <div>
    <ul v-for="(x, i) in books" :key="i">
      <li>{{ x.title }} / {{ x.price }} / {{ categories[i] }}</li>
    </ul>
  </div>
</template>

<script>
import { pipe, andThen, map, prop } from 'ramda'
import { all } from 'wink-fp'

let bookCategories = [
  { id: 1, category: 'FP' },
  { id: 2, category: 'FRP' },
  { id: 3, category: 'JS' }
]

let fetchCategory = async id =>
  bookCategories.find(x => x.id === id).category

let fetchBooks = async () => [
  { title: 'FP in JavaScript', price: 100, categoryId: 1 },
  { title: 'RxJS in Action', price: 200, categoryId: 2 },
  { title: 'Speaking JavaScript', price: 300, categoryId: 3 }
]

let mounted = function() {
  pipe(
    fetchBooks,
    andThen(x => this.books = x),
    andThen(map(pipe(prop('categoryId'), fetchCategory))),
    andThen(all),
    andThen(x => this.categories = x)
  )()
}

export default {
  name:'App',
  data: () => ({
    books: [],
    categories: []
  }),
  mounted,
}
</script>

37 行

fetchBooks,
andThen(x => this.books = x),

呼叫 fetchBooks() API，並即時寫入 books data。

39 行

andThen(map(pipe(prop('categoryId'), fetchCategory))),
andThen(all),
andThen(x => this.categories = x)

在 map() 內呼叫 fetchCategory()，由於回傳 Promise Array，因此需再傳入 Promise.all() 處理，all() 為 Promise.all() 的 funciton 版本，最後寫入 categories data。

RxJS

<template>
  <div>
    <ul v-for="(x, i) in books" :key="i">
      <li>{{ x.title }} / {{ x.price }} / {{ categories[i] }}</li>
    </ul>
  </div>
</template>

<script>
import { from, of } from 'rxjs'
import { concatMap, share, toArray } from 'rxjs/operators'

let bookCategories = [
  { id: 1, category: 'FP' },
  { id: 2, category: 'FRP' },
  { id: 3, category: 'JS' }
]

let fetchCategory$ = id =>
  of(bookCategories.find(x => x.id === id).category)

let fetchBooks$ = () => from([
  { title: 'FP in JavaScript', price: 100, categoryId: 1 },
  { title: 'RxJS in Action', price: 200, categoryId: 2 },
  { title: 'Speaking JavaScript', price: 300, categoryId: 3 }
])

let subscriptions = () => {
  let books$ = fetchBooks$().pipe(share())

  let books = books$.pipe(toArray())

  let categories = books$.pipe(
    concatMap(x => fetchCategory$(x.categoryId)),
    toArray()
  )

  return { books, categories }
}

export default {
  name:'App',
  subscriptions
}
</script>

22 行

let fetchBooks$ = () => from([
  { title: 'FP in JavaScript', price: 100, categoryId: 1 },
  { title: 'RxJS in Action', price: 200, categoryId: 2 },
  { title: 'Speaking JavaScript', price: 300, categoryId: 3 }
])

模擬 RxJS 的 ajax() 從 API 回傳 Observable，由於遵守 RESTFul 設計，因此僅回傳 catetoryId，若要取得 category 必須再呼叫另外一個 API。

from() 可直接將 Array 轉成 Observable

19 行

let fetchCategory$ = id =>
  of(bookCategories.find(x => x.id === id).category)

模擬另外一個 API 由 categoryId 取得 category。

of() 可直接將單一 primitive 轉成 Observable

29 行

let books$ = fetchBooks$().pipe(share())

由 fetchBooks$() 取得 Observable，因為將以 books$ 產生兩個 side effect，為避免出現兩個 API request，因此加上 share() 共用 Observable。

31 行

let books = books$.pipe(toArray())

目前 Vue 的 HTML template 僅支援 Observable 內為 primitive，並不支援 Observable 內為 Array，因此必須使用 toArray() 由 Observable 轉成 array 後，才能在 HTML template 中使用 v-for。

33 行

let categories = books$.pipe(
  concatMap(x => fetchCategory$(x.categoryId)),
  toArray()
)

由 books$() 建立 categories，因為 books$ 本身是 Observable，而 fetchCategory$() 亦回傳 Observable，若使用單純的 map() 勢必造 Observable of Observable。

Observable 並不像 Promise 的 then() 天生具有 chain() 或 flatMap() 特性，必須選擇適當 operator 才能攤平 Observable。

對於這種 nested API 呼叫有兩個選擇： concatMap() 或 flatMap() (亦稱 mergeMap())：

  • concatMap()：合併後順序與 books$ 相同
  • flatMap()：合併後順序與 fetchCategory$ 回傳相同

一般來說，對於 Nested Observable 我們會希望維持第一個 Observable 原本順序，因此使用 concatMap()。

concatMap() 結果與第一個 Observable 的 1、3、5 相同，不受 Observable 出現時間點影響。

mergeMap() 會受 Observable 出現時間點影響，若第二個 API 因網路速度回傳順序不如預期，則會改變最後回傳 Observable 結果。

34 行

return { books, categories }

最後回傳 books 與 categories 兩個 side effect。

28 行

let subscriptions = () => {
}

由於 Observable 都沒使用到 this 存取 data，可安心使用 arrow function。

之前方式都看似不錯，唯動用兩份 data 比較遺憾，可否只使用一份 side effect 呢 ?

Promise Chain

<template>
  <div>
    <ul v-for="(x, i) in books" :key="i">
      <li>{{ x.title }} / {{ x.price }} / {{ x.category }}</li>
    </ul>
  </div>
</template>

<script>
let bookCategories = [
  { id: 1, category: 'FP' },
  { id: 2, category: 'FRP' },
  { id: 3, category: 'JS' }
]

let fetchCategory = async id =>
  bookCategories.find(x => x.id === id).category

let fetchBooks = async () => [
  { title: 'FP in JavaScript', price: 100, categoryId: 1 },
  { title: 'RxJS in Action', price: 200, categoryId: 2 },
  { title: 'Speaking JavaScript', price: 300, categoryId: 3 }
]

let mounted = function() {
  fetchBooks()
    .then(x => Promise.all([x, Promise.all(x.map(y => fetchCategory(y.categoryId)))]))
    .then(([x, y]) => x.map((z, i) => ({ ...z, category: y[i]})))
    .then(x => this.books = x)
}

export default {
  name:'App',
  data: () => ({
    books: [],
  }),
  mounted,
}
</script>

26 行

fetchBooks()
  .then(x => Promise.all([x, Promise.all(x.map(y => fetchCategory(y.categoryId)))]))

由於只有一份 side effect，因此勢必將 fetchBooks() 回傳的 Promise 往下傳，這對 Promise Chain 比較麻煩，要使用 Promise.all() 將兩個 Promise 透過 Array 一起傳到下一個 then() 處理，因此會看到兩個 Promise.all()，外層是為了將 Promise 往下傳，內層是為了處理 Array of Promise。

使用 Promise.all() 將 asynchronous 與 synchronous 一起傳到下一個 then() 為 Promise Chain 慣用手法

28 行

.then(([x, y]) => x.map((z, i) => ({ ...z, category: y[i]})))
.then(x => this.books = x)

將原本 allBooks computed 要整合的 map() 搬到 then() 內實作，因此只需要單一 books data 即可。

RxJS

<template>
  <div>
    <ul v-for="(x, i) in books" :key="i">
      <li>{{ x.title }} / {{ x.price }} / {{ x.category }}</li>
    </ul>
  </div>

<script>
import { from, of } from 'rxjs'
import { concatMap, map, toArray } from 'rxjs/operators'

let bookCategories = [
  { id: 1, category: 'FP' },
  { id: 2, category: 'FRP' },
  { id: 3, category: 'JS' }
]

let fetchCategory$ = id =>

let subscriptions = () => {

</script>

30 行

concatMap(x => fetchCategory$(x.categoryId).pipe(
  map(y => ({ ...x, category: y }))
)),

Promise Chain 遇到傳遞 Promise 問題在 RxJS 不會出現，因為 RxJS 需使用 Nested Pipe 完成，剛好很巧妙地讀取到 outer variable。

concatMap() 在 RxJS 5 時有第二個 argument 可以傳入 function 直接 map()，但 RxJS 6 已經拿掉，官方建議改用 inner map 實現，因此出現了 Nested Pipe。

Conclusion

  • 示範了這麼多種寫法，大多數人應該最能接受 async await 方式，無論 double side effect 或 single side effect 都最直覺易懂
  • 個人則偏好 RxJS 的 double side effect 方式，算是寫法最精簡，且 function pipeline 最漂亮寫法
  • 若有 Nested Promise 需求，其實不必追求 single side effect，這種對 function pipeline 難度較高，畢竟有多個 API request，有多份 side effect 也是天經地義
