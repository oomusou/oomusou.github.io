自行實作 Maybe Monad

2020-04-16

Maybe 也是看似很玄的概念，讓我們不用處理 undefined，且可使用 map() 與 chain() 直接改變 Maybe 內部值，其實只要 30 行左右就可自行實作 Maybe，讓我們對 Maybe 更加了解。

Version

macOS Catalina 10.15.4
VS Code 1.44.0
Quokka 1.0.285
ECMAScript 2015

Method Chaining

class MyMaybe {
  #value = null
  #isNothing = true

  constructor(value) {
    if (!value) 
      this.#isNothing = true
    else {
      this.#isNothing = false
      this.#value = value
    }
  }

  static Just(value) {
    return new MyMaybe(value)
  }

  static Nothing() {
    return new MyMaybe()
  }

  map(f) {
    return this.#isNothing ? MyMaybe.Nothing() : MyMaybe.Just(f(this.#value))
  }

  chain(f) {
    return this.#isNothing ? MyMaybe.Nothing() : f(this.#value)
  }

  get(f) {
    this.#isNothing ? f('Nothing') : f(this.#value)
  }
}

let find = (f, a) => {  
  let result = a.find(f)
  return (result === undefined) ? MyMaybe.Nothing() : MyMaybe.Just(result)
}

let prop = k => o => o[k]

let prop_ = k => o => !o[k] ? MyMaybe.Nothing() : MyMaybe.Just(o[k])

let data = [
  { title: 'FP in JavaScript', price: 100 },
  { title: 'RxJS in Action', price: 200 },
  { title: 'Speaking JavaScript', price: 300 }
]

find(x => x.title === 'Speaking JavaScript', data)
  .map(prop('price'))
  .get(x => console.log(x))

find(x => x.title === 'Speaking JavaScript', data)
  .chain(prop_('price'))
  .get(x => console.log(x))

find(x => x.title === 'RxJS', data)
  .map(prop('price'))
  .get(x => console.log(x))

find(x => x.title === 'RxJS', data)
  .chain(prop_('price'))
  .get(x => console.log(x))

第 2 行

#value = null
#isNothing = true

MyMaybe 內有兩個 private field，#value 負責存包進 Maybe 的原始值，#isNothing 則表示目前為 Nothing 或 Just。

第 5 行

constructor(value) {
  if (!value) 
    this.#isNothing = true
  else {
    this.#isNothing = false
    this.#value = value
  }
}

實現 type constructor，若有傳入值則為 Just，否則為 Nothing。

14 行

static Just(value) {
  return new MyMaybe(value)
}

實現 unit function，使用 MyMaybe constructor 建立 Just 並回傳。

18 行

static Nothing() {
  return new MyMaybe()
}

實現 unit function，使用 MyMaybe constructor 建立 Nothing 並回傳。

22 行

map(f) {
  return this.#isNothing ? MyMaybe.Nothing() : MyMaybe.Just(f(this.#value))
}

實現 bind function，將 Maybe 內的值套用在傳入 function，因為 Nothing 不會執行 map()，因此要先使用 #isNothing 判斷是否為 Nothing，若為 Nothing 則重新建立新 Nothing 回傳，若為 Just 則將 #value 傳入 function，結果再使用 unit function 包一遍才會成為 Maybe。

26 行

chain(f) {
  return this.#isNothing ? MyMaybe.Nothing() : f(this.#value)
}

實現 chain function，將 Maybe 內的值套用在傳入 function，同樣也必須判斷是否為 Nothing，若為 Nothing 則不執行 chain()。

因為傳入 f() 已經回傳 Monad，因此不必再使用 unit function，否則會變成兩層 Monad。

35 行

let find = (f, a) => {  
  let result = a.find(f)
  return (result === undefined) ? MyMaybe.Nothing() : MyMaybe.Just(result)
}

自行實作回傳 Maybe 的 find()，底層仍然使用 ECMAScript 的 find()，若為 undefined 則回傳 Nothing，否則回傳 Just。

40 行

let prop = k => o => o[k]

let prop_ = k => o => !o[k] ? MyMaybe.Nothing() : MyMaybe.Just(o[k])

自行實作取得 object property 的 point-free function，差異是 prop() 直接回傳有可能是 undefined，而 prop_() 則回傳 Maybe。

50 行

find(x => x.title === 'Speaking JavaScript', data)
  .map(prop('price'))
  .get(x => console.log(x))

find(x => x.title === 'Speaking JavaScript', data)
  .chain(prop_('price'))
  .get(x => console.log(x))

由於 MyMaybe 使用 class 實作，且每個 function 都回傳 Maybe，因此適合 method chaining。

title 存在 Speaking JavaScript，因此 find() 回傳 Just。

prop() 只回傳一般值，因此使用 map()；prop_() 回傳 Maybe，因此要使用 chain()。

58 行

find(x => x.title === 'RxJS', data)
  .map(prop('price'))
  .get(x => console.log(x))

find(x => x.title === 'RxJS', data)
  .chain(prop_('price'))
  .get(x => console.log(x))

title 不存在 RxJS，因此 find() 回傳 Nothing。

Function Pipeline

import { pipe } from 'ramda'

class MyMaybe {
  #value = null
  #isNothing = true

  constructor(value) {
    if (!value) 
      this.#isNothing = true
    else {
      this.#isNothing = false
      this.#value = value
    }
  }

  static Just(value) {
    return new MyMaybe(value)
  }

  static Nothing() {
    return new MyMaybe()
  }

  map(f) {
    return this.#isNothing ? MyMaybe.Nothing() : MyMaybe.Just(f(this.#value))
  }

  chain(f) {
    return this.#isNothing ? MyMaybe.Nothing() : f(this.#value)
  }

  get(f) {
    this.#isNothing ? f('Nothing') : f(this.#value)
  }
}

let find = f => a => {  
  let result = a.find(f)
  return (result === undefined) ? MyMaybe.Nothing() : MyMaybe.Just(result)
}

let prop = k => o => o[k]

let prop_ = k => o => !o[k] ? MyMaybe.Nothing() : MyMaybe.Just(o[k])

let map = f => o => o.map(f)

let chain = f => o => o.chain(f)

let get = f => o => o.get(f)

let data = [
  { title: 'FP in JavaScript', price: 100 },
  { title: 'RxJS in Action', price: 200 },
  { title: 'Speaking JavaScript', price: 300 }
]

pipe(
  find(x => x.title === 'Speaking JavaScript'),
  map(prop('price')),
  get(x => console.log(x))
)(data)

pipe(
  find(x => x.title === 'Speaking JavaScript'),
  chain(prop_('price')),
  get(x => console.log(x))
)(data)

pipe(
  find(x => x.title === 'RxJS'),
  map(prop('price')),
  get(x => console.log(x))
)(data)

pipe(
  find(x => x.title === 'RxJS'),
  chain(prop_('price')),
  get(x => console.log(x))
)(data)

之前是以 method chaining 使用 Maybe，我們只要另外提供 point-free 版本的 map()、chain() 與 get()，就可改用 function pipeline。

第 46 行

let map = f => o => o.map(f)

let chain = f => o => o.chain(f)

let get = f => o => o.get(f)

提供 function 版本的 map()、chain() 與 get()。

58 行

pipe(
  find(x => x.title === 'Speaking JavaScript'),
  map(prop('price')),
  get(x => console.log(x))
)(data)

如此就能以 pipe() 組合 find()、map() 與 get()。

Point-free

import { pipe, prop, propEq } from 'ramda'
import { log } from 'wink-fp'

class MyMaybe {
  #value = null
  #isNothing = true

  constructor(value) {
    if (!value) 
      this.#isNothing = true
    else {
      this.#isNothing = false
      this.#value = value
    }
  }

  static Just(value) {
    return new MyMaybe(value)
  }

  static Nothing() {
    return new MyMaybe()
  }

  map(f) {
    return this.#isNothing ? MyMaybe.Nothing() : MyMaybe.Just(f(this.#value))
  }

  chain(f) {
    return this.#isNothing ? MyMaybe.Nothing() : f(this.#value)
  }

  get(f) {
    this.#isNothing ? f('Nothing') : f(this.#value)
  }
}

let find = f => a => {  
  let result = a.find(f)
  return (result === undefined) ? MyMaybe.Nothing() : MyMaybe.Just(result)
}

let prop_ = k => o => !o[k] ? MyMaybe.Nothing() : MyMaybe.Just(o[k])

let map = f => o => o.map(f)

let chain = f => o => o.chain(f)

let get = f => o => o.get(f)

let data = [
  { title: 'FP in JavaScript', price: 100 },
  { title: 'RxJS in Action', price: 200 },
  { title: 'Speaking JavaScript', price: 300 }
]

pipe(
  find(propEq('title', 'Speaking JavaScript')),
  map(prop('price')),
  get(log)
)(data)

pipe(
  find(propEq('title', 'Speaking JavaScript')),
  chain(prop_('price')),
  get(log)
)(data)

pipe(
  find(propEq('title', 'RxJS')),
  map(prop('price')),
  get(log)
)(data)

pipe(
  find(propEq('title', 'RxJS')),
  chain(prop_('price')),
  get(log)
)(data)

第 1 行

import { pipe, prop, propEq } from 'ramda'
import { log } from 'wink-fp'

加入 Ramda 的 Wink-fp 的 function 後，可以使寫法更 point-free。

57 行

pipe(
  find(propEq('title', 'Speaking JavaScript')),
  map(prop('price')),
  get(log)
)(data)

我們自行實作的 Maybe 也能完美使用 function pipeline 與 point-free。

Conclusion

  • 之所以沒用 Crocks 的 prop()，是因為自行實作的 Maybe 並沒有遵行 Fantasy Land 的 Maybe 規格，因此無法使用 Crocks 的 prop()，這也能看出 Fantasy Land 的偉大，使的眾多 FP library 可混搭使用
  • 由本文可發現僅管 Maybe 使用 class 實作，只要提供 point-free 的 function，最後亦可使用 function pipeline，這也是為什麼 Crocks 的 Maybe 同時提供 method chaining 與 function pipeline 兩種使用方式
  • Class 在 FP 並非完全不能使用，把 class 視為 container 負責將 value 包在 container 內，並實踐相對應的 interface，只要搭配輔助的 point-free function，就可繼續使用 function pipeline，重點是從 object composition 轉成 function composition
