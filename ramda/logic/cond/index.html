使用 cond() 實現 Switch Case

2020-02-27

寫程式免不了要使用判斷邏輯，若要實現 switch 根據不同條件執行不同 function，Ramda 提供了 cond()。

Version

macOS Catalina 10.15.3
VS Code 1.42.1
Quokka 1.0.277
Ramda 0.27.0

Imperative

let data = [
  { title: 'FP in JavaScript', price: 100, category: 'FP' },
  { title: 'Rx in Action', price: 200, category: 'FRP' },
  { title: 'Speaking JavaScript', price: 300, category: 'JS' }
]

let f = arr => {
  let result = []

  for (let x of arr) {
    let x_ = Object.assign({}, x)

    switch(x.category) {
      case 'FP':
        x_.price = x.price * 0.8
        break
      case 'FRP':
        x_.price = x.price * 0.7
        break
      default:
        x_.price = x.price * 0.6
    }

    result.push(x_)
  }

  return result
}

f(data) // ?

第 1 行

let data = [
  { title: 'FP in JavaScript', price: 100, category: 'FP' },
  { title: 'Rx in Action', price: 200, category: 'FRP' },
  { title: 'Speaking JavaScript', price: 300, category: 'JS' }
]
  • 若 category 為 FP，price 打 8 折
  • 若 category 為 FRP，price 打 7 折
  • 其餘則打 6 折

第 10 行

for (let x of arr) {
  let x_ = Object.assign({}, x)

  switch(x.category) {
    case 'FP':
      x_.price = x.price * 0.8
      break
    case 'FRP':
      x_.price = x.price * 0.7
      break
    default:
      x_.price = x.price * 0.6
  }

  result.push(x_)
}

Imperative 會先建立 result empty array，使用 for loop 對 data 一筆一筆處理，然後使用 Object.assign() clone object，並使用 switch case 判斷 category，依據各自 case 計算 price，最後 push 進 result array 回傳。

ECMAScript

reduce()

let data = [
  { title: 'FP in JavaScript', price: 100, category: 'FP' },
  { title: 'Rx in Action', price: 200, category: 'FRP' },
  { title: 'Speaking JavaScript', price: 300, category: 'JS' }
]

let f = arr => arr.reduce((a, x) => {
  switch (x.category) {
    case 'FP':
      return [...a, {...x, price: x.price * 0.8 }]
    case 'FRP':
      return [...a, { ...x, price: x.price * 0.7 }]
    default:
      return [...a, { ...x, price: x.price * 0.6 }]
  }
}, [])

f(data) // ?

只要能使用 for loop，就能使用 reduce() 改寫。

Ramda

import { map } from 'ramda'

let data = [
  { title: 'FP in JavaScript', price: 100, category: 'FP' },
  { title: 'Rx in Action', price: 200, category: 'FRP' },
  { title: 'Speaking JavaScript', price: 300, category: 'JS' }
]

let f = map(x => {
  switch (x.category) {
    case 'FP':
      return {...x, price: x.price * 0.8 }
    case 'FRP':
      return { ...x, price: x.price * 0.7 }
    default:
      return { ...x, price: x.price * 0.6 }
  }
})

f(data) // ?

也可使用 Ramda 的 map() 取代，可使 arr argument point-free。

cond()

import { map, compose, multiply, prop, propEq } from 'ramda'
import { T, cond, chain, assoc } from 'ramda'

let data = [
  { title: 'FP in JavaScript', price: 100, category: 'FP' },
  { title: 'Rx in Action', price: 200, category: 'FRP' },
  { title: 'Speaking JavaScript', price: 300, category: 'JS'}
]

let applyDiscount = discount => compose(multiply(discount), prop('price'))

let f = map(cond([
  [propEq('category', 'FP'), chain(assoc('price'), applyDiscount(0.8))],
  [propEq('category', 'FRP'), chain(assoc('price'), applyDiscount(0.7))],
  [T, chain(assoc('price'), applyDiscount(0.6))]
]))

f(data) // ?

Ramda 正規方式是使用 cond() 取代 switch case。

當 propEq() 為 true 時，使用 chain() 與 assoc() 組合改變 price 值。

cond()
[[(*… → Boolean),(*… → *)]] → (*… → *)
根據條件回傳不同 function，相當於 switch case

[[(*… → Boolean),(*… → *)]]：array of array，每個小 array 包含 predicate 與要執行 function

(*… → *)：回傳所選擇 function

另外 T() 相當於 () => true，由於 always true，相當於 switch case 的 default。

Lens

import { map, cond, propEq, identity, lensProp, over, multiply, T } from 'ramda'

let data = [
  { title: 'FP in JavaScript', price: 100, category: 'FP' },
  { title: 'Rx in Action', price: 200, category: 'FRP' },
  { title: 'Speaking JavaScript', price: 300, category: 'JS'}
]

let priceLens = lensProp('price')

let f = map(cond([
  [propEq('category', 'FP'), over(priceLens, multiply(0.8))],
  [propEq('category', 'FRP'), over(priceLens, multiply(0.7))],
  [T, over(priceLens, multiply(0.6))],
]))

f(data) // ?

若要修改 object，透過 lens 是不錯方式。

第 9 行

let priceLens = lensProp('price')

先使用 lensProp() 為 price property 建立 lens。

12 行

[propEq('category', 'FP'), over(priceLens, multiply(0.8))],

直接透過 over 修改 price property，不必搭配 assoc()、chain() 與 prop()。

Conclusion

  • ifElse()、when() 與 unless() 都只適合單一條件，若要多條件就要靠 cond()
