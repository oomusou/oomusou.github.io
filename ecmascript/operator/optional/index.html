ECMAScript 之 Optional Chaining Operator

2020-07-08

ECMAScript 2020 支援了 ?. Optional Chaining Operator，至此 ECMAScript 終於有專屬的 Operator 用於 Nested Object，不必再借用 && 與 Falsy Value。

Version

macOS Catalina 10.15.5
WebStorm 2020.1.2
Vue 2.6.11
ECMAScript 2020

Object

若直接讀取 Object 不存在的 property，會出現 undefined 的 runtime error，這在 ECMAScript 經常出現。

<template>
  <div>
    <button v-stream:click="click$">Get Price</button>
    <div>{{ message1$ }}</div>
    <div>{{ message2$ }}</div>
  </div>
</template>

<script>
import { mapTo } from 'rxjs/operators'

let book = {
  title: 'FP in JavaScript',
  price: {
    amazon: 100,
    google: 200
  }
}

let subscriptions = function() {
  let message1$ = this.click$.pipe(
    mapTo(book.price.amazon)
  )

  let message2$ = this.click$.pipe(
    mapTo(book.fee.amazon)
  )

  return { message1$, message2$ }
}

export default {
  name:'App',
  domStreams: [
    'click$'
  ],
  subscriptions
}
</script>

12 行

let book = {
  title: 'FP in JavaScript',
  price: {
    amazon: 100,
    google: 200
  }
}

book Object 下有 price property，但沒有 fee property。

25 行

let message2$ = this.click$.pipe(
  mapTo(book.fee.amazon)
)

若試圖讀取 book Object 下的 fee property，會造成 undefined runtime error。

&&

搭配 && 檢查 property 是否存在避免 undefined runtime error。

<template>
  <div>
    <button v-stream:click="click$">Get Price</button>
    <div>{{ message1$ }}</div>
    <div>{{ message2$ }}</div>
  </div>
</template>

<script>
import { mapTo } from 'rxjs/operators'

let book = {
  title: 'FP in JavaScript',
  price: {
    amazon: 100,
    google: 200
  }
}

let subscriptions = function() {
  let message1$ = this.click$.pipe(
    mapTo(book.price && book.price.amazon)
  )

  let message2$ = this.click$.pipe(
    mapTo(book.fee && book.fee.amazon)
  )

  return { message1$, message2$ }
}

export default {
  name:'App',
  domStreams: [
    'click$'
  ],
  subscriptions
}
</script>

25 行

let message2$ = this.click$.pipe(
  mapTo(book.fee && book.fee.amazon)
)

傳統會先使用 && 判斷 book.fee property 是否存在，若存在才讀取該 property。

如 fee Object 根本不存在，則 && 右側不會執行，直接 short-circuiting 回傳 undefined。

但 && 其實是判斷 truthy value，而 1、non-empty string 與 true 皆是 truthy value，因此可能誤判，並不是最完美解法。

?.

也能搭配 ?. 檢查 property 是否存在避免 undefined runtime error。

<template>
  <div>
    <button v-stream:click="click$">Get Price</button>
    <div>{{ message1$ }}</div>
    <div>{{ message2$ }}</div>
  </div>
</template>

<script>
import { mapTo } from 'rxjs/operators'

let book = {
  title: 'FP in JavaScript',
  price: {
    amazon: 100,
    google: 200
  }
}

let subscriptions = function() {
  let message1$ = this.click$.pipe(
    mapTo(book.price?.amazon)
  )

  let message2$ = this.click$.pipe(
    mapTo(book.fee?.amazon)
  )

  return { message1$, message2$ }
}

export default {
  name:'App',
  domStreams: [
    'click$'
  ],
  subscriptions
}
</script>

25 行

let message2$ = this.click$.pipe(
  mapTo(book.fee?.amazon)
)

ES2020 支援了 ?.，若 book.fee property 存在才會讀取 book.fee.amazon，否則直接回傳 undefined。

?. 只判斷 null 與 undefined 而非 truthy value，因此不用擔心誤判。

Function

Check Function Exist

判斷 function 存在才呼叫，因此能避免 runtime error。

<template>
  <div>
    <button v-stream:click="click$">Check Function Exist</button>
    <div>{{ message1$ }}</div>
    <div>{{ message2$ }}</div>
  </div>
</template>

<script>
import { mapTo } from 'rxjs/operators'

let book = {
  title: 'FP in JavaScript',
  price: 100,
  showDescription: function() {
    return `${this.title} : ${this.price}`
  }
}

let subscriptions = function() {
  let message1$ = this.click$.pipe(
    mapTo(book.showDescription?.())
  )

  let message2$ = this.click$.pipe(
    mapTo(book.show?.())
  )

  return { message1$, message2$ }
}

export default {
  name:'App',
  domStreams: [
    'click$'
  ],
  subscriptions
}
</script>

12 行

let book = {
  title: 'FP in JavaScript',
  price: 100,
  showDescription: function() {
    return `${this.title} : ${this.price}`
  }
}

book Object 只存在 showDescription()，並不存在 show()。

25 行

let message2$ = this.click$.pipe(
  mapTo(book.show?.())
)

若呼叫不存在的 show() 會直接 runtime error，傳統也會使用 && 檢查 method 是否存在才呼叫，ES2020 可在 method 名稱之後加上 ?.()，若 method 不存在則回傳 undefined，若存在則執行該 method 回傳結果。

Check Function Argument

若傳入 function 則印出 log。

<template>
  <div>
    <button v-stream:click="click$">Check Function Argument</button>
    <div>{{ message1$ }}</div>
    <div>{{ message2$ }}</div>
  </div>
</template>

<script>
import { map } from 'rxjs/operators'

let book = {
  title: 'FP in JavaScript',
  price: 100,
  showDescription: function(log) {
    let description = `${this.title} : ${this.price}`
    log?.(description)
    return description
  }
}

let subscriptions = function() {
  let message1$ = this.click$.pipe(
    map(_ => book.showDescription())
  )

  let message2$ = this.click$.pipe(
    map(_ => book.showDescription(console.log))
  )

  return { message1$, message2$ }
}

export default {
  name:'App',
  domStreams: [
    'click$'
  ],
  subscriptions
}
</script>

實務上若 function 的傳入為 optional，則可使用 ?. 判斷，若有傳入則執行，否則不執行。

12 行

let book = {
  title: 'FP in JavaScript',
  price: 100,
  showDescription: function(log) {
    let description = `${this.title} : ${this.price}`
    log?.(description)
    return description
  }
}

log argument 為 optional，可使用 ?. 判斷是否存在，若存在則呼叫 function，否則不呼叫。

23 行

let message1$ = this.click$.pipe(
  map(_ => book.showDescription())
)

showDescription() 沒傳入任何值，因此不會印出任何值。

27 行

let message2$ = this.click$.pipe(
  map(_ => book.showDescription(console.log))
)

showDescription() 傳入 console.log()，因此會印出 log。

Array

判斷 Array 是否存在，存在則印出，否則回傳 undefined。

<template>
  <div>
    <button v-stream:click="click$">Show Array</button>
    <div>{{ message1$ }}</div>
    <div>{{ message2$ }}</div>
  </div>
</template>

<script>
import { mapTo } from 'rxjs/operators'

let foo = [1, 2, 3]
let too = undefined

let subscriptions = function() {
  let message1$ = this.click$.pipe(
    mapTo(foo?.[1])
  )

  let message2$ = this.click$.pipe(
    mapTo(too?.[1])
  )

  return { message1$, message2$ }
}

export default {
  name:'App',
  domStreams: [
    'click$'
  ],
  subscriptions
}
</script>

12 行

let foo = [1, 2, 3]
let too = undefined

foo 為 Array，但 too 並不是。

20 行

let message2$ = this.click$.pipe(
  mapTo(too?.[1])
)

若 too 是 Array 才取其 index 1，否則回傳 undefined。

Combine ?. with ??

實務上 ?. 長與 ?? 搭配使用。

<template>
  <div>
    <button v-stream:click="click$">Combine ?. with ??</button>
    <div>{{ message$ }}</div>
  </div>
</template>

<script>
import { mapTo } from 'rxjs/operators'

let book = {
  title: 'FP in JavaScript',
  price: {
    amazon: 100,
    google: 200
  }
}

let subscriptions = function() {
  let message$ = this.click$.pipe(
    mapTo(book?.name ?? 'Anonymous')
  )

  return { message$ }
}

export default {
  name:'App',
  domStreams: [
    'click$'
  ],
  subscriptions
}
</script>

11 行

let book = {
  title: 'FP in JavaScript',
  price: {
    amazon: 100,
    google: 200
  }
}

book Object 並不存在 name property。

20 行

let message$ = this.click$.pipe(
  mapTo(book?.name ?? 'Anonymous')
)

?. 判斷 name property 是否存在，因不存在所以回傳 undefined。

?? 判斷左側為 undefined，因此回傳右側的 Anonymous。

Error.response

當 API 回傳 Rejected Promise 時，顯示其 HTTP status code。

<template>
  <div>
    <button @click="onClick">Error.response</button>
    <div>{{ message }}</div>
  </div>
</template>

<script>
import axios from 'axios'

let fetchAPI = _ => axios.get('http://localhost:3000/')

let onClick = function() {
  fetchAPI()
    .then(x => x.data)
    .then(x => this.message = x)
    .catch(e => console.log(e.response?.status??'404'))
}

export default {
  name:'App',
  data: () => ({
    message: ''
  }),
  methods: {
    onClick
  }
}
</script>

14 行

fetchAPI()
  .then(x => x.data)
  .then(x => this.message = x)
  .catch(e => console.log(e.response?.status??'404'))

若要讀取 HTTP status code，可在 .catch() 讀取 x.response.status，但問題是 response 並不是永遠存在，如斷網時，error object 並沒有 response Object。

  • 可使用 ?. 判斷 response object 是否存在，若存在則回傳 status property
  • 若 response object 不存在為 undefined，可用 ?? 提供預設值 404

Conclusion

  • ?. 為 ES2020 標準，目前 browser 都已支援，front-end 可安心使用
  • ?. 支援 short-circuiting，如 ?. 左側是 null 或 undefined，則 ?. 將不會執行
  • ?. 常與 ?? 一同搭配使用，?. 判斷 property 是否存在，?? 提供 undefined 的預設值
  • ?. 實務上最常用在 Error.reponse，因為 response Object 並不是永遠存在，並搭配 ?? 提供預設值
