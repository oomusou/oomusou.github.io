Monad-tw 之 NavbarDesktop Component 解析

2020-10-18

NavbarDesktop Component 為 NavBar Component 內負責在 Desktop 顯示的實質 Navbar。

Version

Monad-tw 0.0.1

NavbarDesktop Component

NavbarDesktop component 負責顯示 desktop 的 navbar、search 與 dark mode toggle。

components/Navbar/NavbarDesktop.vue

<template>
  <nav class="hidden md:block container mx-auto dark:text-gray-400 py-3">
    <div class="flex items-center mx-4">
      <!-- title -->
      <div class="mr-6">
        <span class="font-semibold text-xl tracking-tight">
          {{ title }}
        </span>
      </div>

      <!-- menu -->
      <div class="flex-grow">
        <ul class="flex">
          <li v-for="navItem in menu" :key="navItem.name" class="px-4 py-1">

            <!-- internal link -->
            <g-link v-if="isShowInternalLink(navItem)" class="inline-block py-1" :to="navItem.link" :title="navItem.name">
              {{ navItem.name }}
            </g-link>

            <!-- external link -->
            <a v-if="isShowExternalLink(navItem)" class="inline-block py-1" :href="navItem.link" target="_blank" :title="navItem.name">
              {{ navItem.name }}
            </a>

            <!-- More -->
            <client-only>
              <v-popover v-if="isShowMore(navItem)" placement="top" popoverClass="navbar-popover" offset="16">
                <a class="inline-block py-1 cursor-pointer">
                  {{ navItem.name }}
                  <font-awesome :icon="['fas', 'angle-down']"/>
                </a>

                <!-- More submenu -->
                <template slot="popover">
                  <ul>
                    <li v-for="subItem in navItem.children" :key="subItem.name" class="px-4 py-2 submenu-item hover:text-white">
                      <!-- internal link -->
                      <g-link v-if="isShowSubMenuInternalLink(subItem)" class="inline-block" :to="subItem.link" :title="subItem.name">
                        {{ subItem.name }}
                      </g-link>

                      <!-- external link -->
                      <a v-if="isShowSubMenuExternalLink(subItem)" class="inline-block" :href="subItem.link" target="_blank" :title="subItem.name">
                        {{ subItem.name }}
                      </a>
                    </li>
                  </ul>
                </template>
              </v-popover>
            </client-only>
          </li>

          <!-- ... -->
          <li class="px-4 py-1">
            <a role="button" :class="showBlueText" class="inline-block px-4 py-1" @click.prevent="onToggleSubNavigation" aria-label="Open Subnavigation" title="Open Subnavigation">
              <font-awesome :icon="['fas', 'ellipsis-h']" size="lg"/>
            </a>

            <!-- ... sub-navigation -->
            <div :class="showSubNavigation" class="mega-menu bg-white border-t border-gray-200 shadow-xl mb-16 py-4 dark:bg-black dark:border-gray-900" v-click-outside="onClickOutside">
              <sub-navigation/>
            </div>
          </li>
        </ul>
      </div>

      <!-- search & toggle -->
      <div>
        <ul class="flex">
          <!-- search -->
          <li class="mr-6">
            <search-button v-on="$listeners"/>
          </li>

          <!-- toggle dark mode -->
          <li>
            <theme-switcher v-on="$listeners" :theme="theme"/>
          </li>
        </ul>
      </div>
    </div>
  </nav>
</template>

<script>
import { pipe, prop, path, allPass, not, __, gt, equals } from 'ramda'
import ThemeSwitcher from '@/components/Navbar/ThemeSwitcher'
import SearchButton from '@/components/Navbar/SearchButton'
import SubNavigation from '@/components/Navbar/NavbarSubNavigation'
import vClickOutside from 'v-click-outside'

// isShowInternalLink :: Object -> Boolean
let isShowInternalLink = allPass([
  pipe(prop('external'), not),
  pipe(path(['children', 'length']), equals(0))
])

// isShowExternalLink :: Object -> Boolean
let isShowExternalLink = allPass([
  prop('external'),
  pipe(path(['children', 'length']), equals(0))
])

// isShowSubMenuInternalLink :: Object -> Boolean
let isShowSubMenuInternalLink = pipe(
  prop('external'),
  not
)

// isShowSubMenuExternalLink :: Object -> Boolean
let isShowSubMenuExternalLink = prop('external')

// isShowMore :: Object -> Boolean
let isShowMore = pipe(
  path(['children', 'length']),
  gt(__, 0)
)

let showBlueText = function() {
  return { 'text-blue-600' : this.isShowSubNavigation }
}

let showSubNavigation = function() {
  return { 'hidden' : !this.isShowSubNavigation }
}

let onToggleSubNavigation = function() {
  this.isShowSubNavigation = !this.isShowSubNavigation
}

let onClickOutside = function(event) {
  if (!event.defaultPrevented && this.isShowSubNavigation)
    this.isShowSubNavigation = false
}

let isShowSubNav_ = function() {
  if (!this.isShowSubNav)
    this.isShowSubNavigation = false
}

let route_ = function() {
  this.isShowSubNavigation = false
}

let created = function() {
  this.title = this.$static.metadata.siteName
  this.menu = this.$static.metadata.headerNavigation
}

export default {
  components: {
    ThemeSwitcher,
    SearchButton,
    SubNavigation
  },
  directives: {
    clickOutside: vClickOutside.directive
  },
  props: {
    theme: { default: '' },
    isShowSubNav: { default: false }
  },
  data: () => ({
    title: '',
    menu: [],
    isShowSubNavigation: false,
    vcoConfig: {
      events: ['dblclick', 'click'],
      isActive: true
    }
  }),
  computed: {
    showBlueText,
    showSubNavigation,
  },
  methods: {
    isShowInternalLink,
    isShowExternalLink,
    isShowSubMenuInternalLink,
    isShowSubMenuExternalLink,
    isShowMore,
    onToggleSubNavigation,
    onClickOutside,
  },
  watch: {
    isShowSubNav: isShowSubNav_,
    $route: route_
  },
  created
}
</script>

<!--@formatter:off-->
<static-query>
query {
  metadata {
    siteName
    headerNavigation {
      name
      link
      external
      children {
        name
        link
        external
      }
    }
  }
}
</static-query>
<!--@formatter:on-->

第 2 行

<nav class="hidden md:block container mx-auto dark:text-gray-400 py-3">

NavbarDesktop component 主要顯示 navbar，因此使用了 HTML 5 的 <nav> 語意較佳，其本質為 block element。

  • hidden md:block：NavbarDesktop component 主要顯示於 desktop，因此一開始為 hidden，只有在 md 以上才會以 block 顯示，此為 Tailwind RWD 慣用技巧
  • container mx-auto：nav 根據不同 breakpoint 有其 max-width，且會自動水平置中
  • dark: text-gray-400：dark mode 下文字為灰色
  • py-3：設定垂直 padding

第 3 行

<div class="flex items-center mx-4">
  • flex items-center：使 item 垂直置中，主要為了 search 與 theme button 能垂直置中
  • mx-4：設定水平 margin

第 4 行

<!-- title -->
<div class="mr-6">
  <span class="font-semibold text-xl tracking-tight">
    {{ title }}
  </span>
</div>
  • mr-6：設定水平 right margin
  • font-semibold：設定字型粗細
  • text-xl：設定字型大小
  • tracking-light：設定字型間距

11 行

<!-- menu -->
<div class="flex-grow">
  • flex-grow：主要顯示 menu，故使用 flex-grow 盡量瓜分剩餘 width

13 行

<ul class="flex">
  • 使用 <ul> 與 <li> 顯示 menu
  • flex：<li> 預設垂直顯示，在此為了實作 menu，故加上 flex 使之水平顯示

14 行

<li v-for="navItem in menu" :key="navItem.name" class="px-4 py-1">

使用 v-for 一一顯示 menu。

16 行

<!-- internal link -->
<g-link v-if="isShowInternalLink(navItem)" class="inline-block py-1" :to="navItem.link" :title="navItem.name">
  {{ navItem.name }}
</g-link>

若 menu item 為內部連結，則使用 Gridsome 的 <g-link>：

  • inline-block：將 <g-link> 本質為 <a>，也就是 inline element，為了要使用 py-1 調整 padding，因此使用 inline-block 使其能使用 padding
  • 使用 isShowInternalLink() 判斷是否為內部連結

21 行

<!-- external link -->
<a v-if="isShowExternalLink(navItem)" class="inline-block py-1" :href="navItem.link" target="_blank" :title="navItem.name">
  {{ navItem.name }}
</a>

若 menu item 為外部連結，則直接使用 <a>：

  • inline-block： <a> 為 inline element，因此使用 inline-block 使其能搭配 py-1 調整 padding
  • 使用 isShowExternalLink() 判斷是否為外部連結

26 行

<!-- More -->
<client-only>
  <v-popover>
    <a>{{ navItem.name }}</a>
    <!-- More submenu -->
    <template slot="popover">
      <ul>
        <li v-for="subItem in navItem.children">
          <!-- internal link -->
          <g-link>{{ subItem.name }}</g-link>

          <!-- external link -->
          <a>{{ subItem.name }}</a>
        </li>
      </ul>
    </template>
  </v-popover>
</client-only>

顯示 More 與其 submenu。

26 行

<!-- More -->
<client-only>
  <v-popover>
  </v-popover>
</client-only>

More 按下以 <v-popover/> component 顯示 submenu，由於 <v-popover> component 不是以 JAMStack 設計，因此要包在 <client-only> 內。

29 行

<a class="inline-block py-1 cursor-pointer">
  {{ navItem.name }}
  <font-awesome :icon="['fas', 'angle-down']"/>
</a>

顯示 More 與其 icon。

More 會在 static query 中取得

34 行

<!-- More submenu -->

  <ul>
    <li v-for="subItem in navItem.children">
      <!-- internal link -->
      <g-link v-if="isShowSubMenuInternalLink(subItem)">
        {{ subItem.name }}
      </g-link>

      <!-- external link -->
      <a v-if="isShowSubMenuExternalLink(subItem)">
        {{ subItem.name }}
      </a>
    </li>
  </ul>
</template>
  • More 的 submenu 以 slot 形式提供給 <v-popover> component 顯示
  • 依舊以 <ul> 與 <li> 實現 submenu
  • 同時提供 <g-link> 與 <a> 版本顯示 submenu，並依照 v-if 判斷顯示

54 行

<!-- ... -->

  </a>

... 與其 sub-navigation。

56 行

<a role="button" :class="showBlueText()" class="inline-block px-4 py-1" @click.prevent="onToggleSubNavigation" aria-label="Open Subnavigation" title="Open Subnavigation">
  <font-awesome :icon="['fas', 'ellipsis-h']" size="lg"/>
</a>
  • :class="showBlueText()"：... 按下去會顯示 藍色，以 showBlueText() 與 class binding 表示
  • inline-block： <a> 為 inline element，因此使用 inline-block 使其能搭配 px-4 py-1 調整 padding

60 行

<!-- ... sub-navigation -->
<div :class="showSubNavigation()" class="mega-menu bg-white border-t border-gray-200 shadow-xl mb-16 py-4 dark:bg-black dark:border-gray-900" v-click-outside="onClickOutside">
  <sub-navigation/>
</div>
  • :class="showSubNavigation()"：... 的 submenu 並不是使用 popup，只是將 <div> 以 hidden 隱藏與顯示而已
  • mega-menu：在 main.scss 自訂的 utility

  • v-click-outside="onClickOutside"：當點擊 <div> 以外的地方會觸發 v-click-outside event，此由 v-click-outside package 所提供 v-click-outside directive

  • Submenu 以 SubNavigation component 顯示

  </li>

data: () => ({

  • title：儲存 site name 供顯示
  • menu：儲存整個 menu
  • isShowSubNavigation：控制是否顯示 ... 按下所顯示的 sub-navigation
  • vcoConfig： v-click-outside package 所使用，當在 <div> 外 double click 與 click 都會觸發 v-click-outside event

146 行

let created = function() {
  this.title = this.$static.metadata.siteName
  this.menu = this.$static.metadata.headerNavigation

一開始在 created hook 從 static query 讀出 siteName 與 headerNavigation 寫進 title 與 menu。

let route_ = function() {
  this.isShowSubNavigation = false
}

route_ 為 $route 的 watch，幫 route 改變時，... 的 SubNavigation component 將隱藏。

137 行

let isShowSubNav_ = function() {
  if (!this.isShowSubNav)
    this.isShowSubNavigation = false
}

isShowSubNav prop 的 watch，當傳入 false 時，... 的 SubNavigation component 將隱藏。

132 行

let onClickOutside = function(event) {
  if (!event.defaultPrevented && this.isShowSubNavigation)
    this.isShowSubNavigation = false

當在 sub-navigation 外 click 或 double click 時，將隱藏 SubNavigation component。

128 行

let onToggleSubNavigation = function() {
  this.isShowSubNavigation = !this.isShowSubNavigation

當按下 ... 時將 toggle SubNavigation component。

120 行

let showBlueText = function() {

showBlueText computed，負責與 ... 的 class binding，根據 isShowSubNavigation state 動態改變 ... 顏色。

let showSubNavigation = function() {
  return { 'hidden' : !this.isShowSubNavigation }

showSubNavigation computed，負責與 SubNavigation component 的 class binding，根據 isShowSubNavigation state 動態決定是否顯示 SubNavigation component。

受否顯示 More，若 static query 回傳的 children array 有內容就會顯示 More。

若 external prop 為 true 則使用 <a> 顯示 More submenu 的 external link。

若 external prop 為 false 則使用 <g-link> 顯示 More submenu 的 internal link。

// isShowInternalLink :: Object -> Boolean

若 external prop 為 true 且 children.length 為 0，也就是不是 submenu，則使用 <a> 顯示 external link。
